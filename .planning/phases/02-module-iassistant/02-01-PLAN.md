**API Route** (`apps/web/src/app/api/prompts/route.ts`):

Create a GET endpoint that fetches all public prompts from the database. No auth required for reading public prompts (they're visible to all authenticated users per RLS policy, but the endpoint can be public since prompts are not sensitive).

```typescript
import { NextResponse } from 'next/server'
import { createClient } from '@/lib/supabase/server'

export async function GET() {
  const supabase = await createClient()

  const { data, error } = await supabase
    .from('prompts')
    .select('id, title, description, llm_provider, model_id, task_type, scope, is_public, category')
    .eq('is_public', true)
    .order('category', { ascending: true })
    .order('title', { ascending: true })

  if (error) {
    return NextResponse.json({ error: 'Failed to load prompts' }, { status: 500 })
  }

  return NextResponse.json({ prompts: data })
}
```

Note: Select only the fields needed by the UI (not prompt_template or system_prompt -- those are server-side only).
